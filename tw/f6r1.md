    function 放置红色羊毛货物 (数量: number) {
        agent.move(RIGHT, 1)
        agent.setItem(RED_WOOL, 数量, 1)
        for (let index = 0; index < 数量; index++) {
            agent.place(FORWARD)
            agent.move(UP, 1)
        }
        agent.move(DOWN, 数量)
    }
    放置红色羊毛货物(1)
